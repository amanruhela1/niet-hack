import pandas as pd import matplotlib.pyplot as plt # Read data from CSV file df = pd.read_csv('recovery_data.csv') # Create a bar chart for percentage of infrastructure restored fig1, ax1 = plt.subplots() ax1.bar(df['City'], df['Infrastructure Restored']) ax1.set_title('Percentage of Infrastructure Restored by City') ax1.set_xlabel('City') ax1.set_ylabel('Percentage') # Create a line chart for recovery rate fig2, ax2 = plt.subplots() ax2.plot(df['Date'], df['Recovery Rate']) ax2.set_title('Recovery Rate Over Time') ax2.set_xlabel('Date') ax2.set_ylabel('Recovery Rate') # Create a scatter plot for recovery time objective fig3, ax3 = plt.subplots() ax3.scatter(df['Cost to Government'], df['Recovery Time Objective']) ax3.set_title('Cost to Government vs. Recovery Time Objective') ax3.set_xlabel('Cost to Government') ax3.set_ylabel('Recovery Time Objective') # Create a pie chart for cost to government for recovery fig4, ax4 = plt.subplots() ax4.pie(df['Cost to Government'], labels=df['City'], autopct='%1.1f%%') ax4.set_title('Cost to Government for Recovery by City') # Display all visualizations fig1
